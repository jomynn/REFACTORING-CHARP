public class LoanService
{
    private const int    MinAge          = 18;
    private const int    MinCreditScore  = 650;
    private const decimal MinIncomeRatio = 0.3m;
    private const decimal MaxLoanAmount  = 1_000_000m;

    public string ApproveLoan(
        User user, LoanRequest request)
    {
        // guard clauses — fail fast, flat structure
        if (user    == null) return "User not found";
        if (!user.IsActive)  return "Inactive user";
        if (user.Age < MinAge) return "Underage";

        if (request == null) return "No request";
        if (request.Amount <= 0)
            return "Invalid amount";
        if (request.Amount > MaxLoanAmount)
            return "Over limit";

        // business rules — now readable
        if (!HasSufficientCreditScore(user))
            return "Low credit";
        if (!HasSufficientIncome(user, request))
            return "Low income";

        return "Approved";
    }

    private bool HasSufficientCreditScore(User user)
        => user.CreditScore >= MinCreditScore;

    private bool HasSufficientIncome(
        User user, LoanRequest request)
        => user.Income >= request.Amount
            * MinIncomeRatio;
}

// AFTER:  flat, named constants, extracted methods

// Business rules readable at a glance

// HasSufficientCreditScore / HasSufficientIncome
